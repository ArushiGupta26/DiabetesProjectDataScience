#K-fold cross-validation method 
from sklearn.model_selection import KFold
names = []
scores = []
for name, model in models:
    
    kfold = KFold(n_splits=10) 
    score = cross_val_score(model, X_train, y_train, cv=kfold, scoring='accuracy').mean()
    names.append(name)
    scores.append(score)
